const { useState, useEffect } = React;

const WORKER_URL = 'https://macro-tracker-proxy.jasontclow.workers.dev';

const GOALS = { calories: 1800, protein: 70, carbs: 240, fat: 60 };

const COLOR = {
  calories: { bar: "#D85A30", text: "#993C1D", light: "#FAECE7" },
  protein:  { bar: "#1D9E75", text: "#0F6E56", light: "#E1F5EE" },
  carbs:    { bar: "#378ADD", text: "#185FA5", light: "#E6F1FB" },
  fat:      { bar: "#BA7517", text: "#854F0B", light: "#FAEEDA" },
};

const todayKey = () => new Date().toISOString().slice(0, 10);

function loadStorage() {
  try {
    const raw = localStorage.getItem("macrotracker_v1");
    return raw ? JSON.parse(raw) : { history: {} };
  } catch { return { history: {} }; }
}

function saveStorage(data) {
  try { localStorage.setItem("macrotracker_v1", JSON.stringify(data)); } catch {}
}

function MacroBar({ label, current, goal, color }) {
  const pct = Math.min(100, Math.round((current / goal) * 100));
  const over = current > goal;
  return (
    <div style={{ marginBottom: 12 }}>
      <div style={{ display: "flex", justifyContent: "space-between", fontSize: 13, marginBottom: 4 }}>
        <span style={{ fontWeight: 500, color: "var(--color-text-primary)" }}>{label}</span>
        <span style={{ color: over ? "#A32D2D" : "var(--color-text-secondary)" }}>
          {Math.round(current)}{label === "Calories" ? "" : "g"} / {goal}{label === "Calories" ? " kcal" : "g"}
        </span>
      </div>
      <div style={{ background: "var(--color-bg-secondary)", borderRadius: 99, height: 10, overflow: "hidden" }}>
        <div style={{
          width: pct + "%", height: "100%", borderRadius: 99,
          background: over ? "#E24B4A" : color.bar,
          transition: "width 0.4s ease"
        }} />
      </div>
      <div style={{ fontSize: 11, color: over ? "#A32D2D" : "var(--color-text-tertiary)", marginTop: 2, textAlign: "right" }}>
        {over ? `${Math.round(current - goal)} over goal` : `${pct}%`}
      </div>
    </div>
  );
}

function FoodItem({ item, onDelete }) {
  return (
    <div style={{
      display: "flex", alignItems: "center", gap: 10,
      padding: "10px 14px", background: "var(--color-bg)",
      border: "0.5px solid var(--color-border)",
      borderRadius: "var(--radius-md)", marginBottom: 8
    }}>
      <div style={{ flex: 1 }}>
        <div style={{ fontWeight: 500, fontSize: 14, color: "var(--color-text-primary)" }}>{item.name}</div>
        <div style={{ fontSize: 12, color: "var(--color-text-secondary)", marginTop: 2 }}>
          {Math.round(item.calories)} kcal · P {Math.round(item.protein)}g · C {Math.round(item.carbs)}g · F {Math.round(item.fat)}g
        </div>
      </div>
      <button onClick={onDelete} style={{
        background: "none", border: "none", cursor: "pointer",
        color: "var(--color-text-tertiary)", fontSize: 16, lineHeight: 1, padding: "2px 6px",
        borderRadius: "var(--radius-md)"
      }}>✕</button>
    </div>
  );
}

function HistoryDay({ date, items }) {
  const totals = items.reduce((a, i) => ({
    calories: a.calories + i.calories, protein: a.protein + i.protein,
    carbs: a.carbs + i.carbs, fat: a.fat + i.fat
  }), { calories: 0, protein: 0, carbs: 0, fat: 0 });

  const [open, setOpen] = useState(false);
  const label = new Date(date + "T12:00:00").toLocaleDateString("en-AU", { weekday: "short", day: "numeric", month: "short" });

  return (
    <div style={{
      background: "var(--color-bg)", border: "0.5px solid var(--color-border)",
      borderRadius: "var(--radius-md)", marginBottom: 10, overflow: "hidden"
    }}>
      <div onClick={() => setOpen(o => !o)} style={{
        display: "flex", justifyContent: "space-between", alignItems: "center",
        padding: "12px 14px", cursor: "pointer"
      }}>
        <div>
          <div style={{ fontWeight: 500, fontSize: 14, color: "var(--color-text-primary)" }}>{label}</div>
          <div style={{ fontSize: 12, color: "var(--color-text-secondary)", marginTop: 2 }}>
            {Math.round(totals.calories)} kcal · P {Math.round(totals.protein)}g · C {Math.round(totals.carbs)}g · F {Math.round(totals.fat)}g
          </div>
        </div>
        <span style={{ fontSize: 12, color: "var(--color-text-tertiary)" }}>
          {open ? "▲" : "▼"} {items.length} item{items.length !== 1 ? "s" : ""}
        </span>
      </div>
      {open && (
        <div style={{ padding: "0 14px 12px", borderTop: "0.5px solid var(--color-border)" }}>
          {items.map((it, i) => (
            <div key={i} style={{
              fontSize: 13, padding: "6px 0",
              borderBottom: i < items.length - 1 ? "0.5px solid var(--color-border)" : "none",
              color: "var(--color-text-primary)"
            }}>
              <span style={{ fontWeight: 500 }}>{it.name}</span>
              <span style={{ color: "var(--color-text-secondary)", marginLeft: 8 }}>
                {Math.round(it.calories)} kcal · P {Math.round(it.protein)}g · C {Math.round(it.carbs)}g · F {Math.round(it.fat)}g
              </span>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

function App() {
  const [tab, setTab] = useState("today");
  const [data, setData] = useState(loadStorage);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState("");

  const today = todayKey();
  const todayItems = data.history[today] || [];
  const totals = todayItems.reduce((a, i) => ({
    calories: a.calories + i.calories, protein: a.protein + i.protein,
    carbs: a.carbs + i.carbs, fat: a.fat + i.fat
  }), { calories: 0, protein: 0, carbs: 0, fat: 0 });

  useEffect(() => { saveStorage(data); }, [data]);

  async function addFood() {
    if (!input.trim()) return;
    setLoading(true);
    setError("");
    try {
      const res = await fetch(WORKER_URL, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-haiku-4-5-20251001",
          max_tokens: 300,
          system: `You are a macro nutrition assistant. When given a food description, respond ONLY with a JSON object (no markdown, no extra text) with these fields: name (string, cleaned-up food name max 40 chars), calories (number), protein (number in grams, 1 decimal), carbs (number in grams, 1 decimal), fat (number in grams, 1 decimal). Base estimates on typical serving sizes if not specified. Example: {"name":"2 scrambled eggs","calories":182,"protein":12.0,"carbs":2.0,"fat":14.0}`,
          messages: [{ role: "user", content: input.trim() }]
        })
      });
      const json = await res.json();
      const text = json.content?.[0]?.text || "";
      const parsed = JSON.parse(text.replace(/```json|```/g, "").trim());
      setData(prev => {
        const day = prev.history[today] || [];
        return { ...prev, history: { ...prev.history, [today]: [...day, parsed] } };
      });
      setInput("");
    } catch {
      setError("Couldn't estimate macros — check your connection and try again.");
    }
    setLoading(false);
  }

  function deleteItem(idx) {
    setData(prev => {
      const day = [...(prev.history[today] || [])];
      day.splice(idx, 1);
      return { ...prev, history: { ...prev.history, [today]: day } };
    });
  }

  const historyDays = Object.entries(data.history)
    .filter(([d]) => d !== today && (data.history[d] || []).length > 0)
    .sort(([a], [b]) => b.localeCompare(a));

  const TabBtn = ({ id, label }) => (
    <button onClick={() => setTab(id)} style={{
      flex: 1, padding: "10px 0", border: "none", cursor: "pointer", fontSize: 14, fontWeight: 500,
      borderBottom: tab === id ? "2px solid #1D9E75" : "2px solid transparent",
      background: "none", color: tab === id ? "#1D9E75" : "var(--color-text-secondary)"
    }}>{label}</button>
  );

  return (
    <div>
      <div style={{ textAlign: "center", marginBottom: 20, paddingTop: "1rem" }}>
        <div style={{ fontSize: 22, fontWeight: 500, color: "var(--color-text-primary)" }}>Macro Tracker</div>
        <div style={{ fontSize: 13, color: "var(--color-text-secondary)", marginTop: 2 }}>
          {new Date().toLocaleDateString("en-AU", { weekday: "long", day: "numeric", month: "long" })}
        </div>
      </div>

      <div style={{ display: "flex", borderBottom: "0.5px solid var(--color-border)", marginBottom: 20 }}>
        <TabBtn id="today" label="Today" />
        <TabBtn id="history" label="History" />
        <TabBtn id="goals" label="Goals" />
      </div>

      {tab === "today" && (
        <>
          <div style={{
            background: "var(--color-bg-secondary)", borderRadius: "var(--radius-lg)",
            padding: "16px", marginBottom: 20
          }}>
            <MacroBar label="Calories" current={totals.calories} goal={GOALS.calories} color={COLOR.calories} />
            <MacroBar label="Protein" current={totals.protein} goal={GOALS.protein} color={COLOR.protein} />
            <MacroBar label="Carbs" current={totals.carbs} goal={GOALS.carbs} color={COLOR.carbs} />
            <MacroBar label="Fat" current={totals.fat} goal={GOALS.fat} color={COLOR.fat} />
          </div>

          <div style={{ display: "flex", gap: 8, marginBottom: 8 }}>
            <input
              value={input} onChange={e => setInput(e.target.value)}
              onKeyDown={e => e.key === "Enter" && !loading && addFood()}
              placeholder='e.g. "2 eggs and toast" or "bowl of rice"'
              disabled={loading}
              style={{
                flex: 1, padding: "10px 12px", fontSize: 14,
                borderRadius: "var(--radius-md)", border: "0.5px solid var(--color-border-secondary)",
                background: "var(--color-bg)", color: "var(--color-text-primary)", outline: "none"
              }}
            />
            <button onClick={addFood} disabled={loading || !input.trim()} style={{
              padding: "10px 16px", borderRadius: "var(--radius-md)", border: "0.5px solid var(--color-border-secondary)",
              background: loading ? "var(--color-bg-secondary)" : "#1D9E75",
              color: loading ? "var(--color-text-secondary)" : "#fff",
              fontWeight: 500, fontSize: 14, cursor: loading ? "not-allowed" : "pointer",
              whiteSpace: "nowrap"
            }}>
              {loading ? "..." : "Add"}
            </button>
          </div>
          {error && <div style={{ fontSize: 13, color: "#A32D2D", marginBottom: 8 }}>{error}</div>}

          <div style={{ marginTop: 16 }}>
            {todayItems.length === 0
              ? <div style={{ textAlign: "center", color: "var(--color-text-tertiary)", fontSize: 14, padding: "32px 0" }}>
                  No food logged today yet — start typing above!
                </div>
              : todayItems.map((item, i) => <FoodItem key={i} item={item} onDelete={() => deleteItem(i)} />)
            }
          </div>
        </>
      )}

      {tab === "history" && (
        <div>
          {historyDays.length === 0
            ? <div style={{ textAlign: "center", color: "var(--color-text-tertiary)", fontSize: 14, padding: "32px 0" }}>
                No history yet — log some food today to get started!
              </div>
            : historyDays.map(([date, items]) => <HistoryDay key={date} date={date} items={items} />)
          }
        </div>
      )}

      {tab === "goals" && (
        <div style={{ background: "var(--color-bg-secondary)", borderRadius: "var(--radius-lg)", padding: 16 }}>
          <div style={{ fontSize: 14, color: "var(--color-text-secondary)", marginBottom: 16 }}>
            Daily targets set for a healthy, active 13-year-old girl. Adjust with guidance from a parent or doctor.
          </div>
          {[
            { label: "Calories", val: GOALS.calories, unit: "kcal", color: COLOR.calories },
            { label: "Protein",  val: GOALS.protein,  unit: "g",    color: COLOR.protein  },
            { label: "Carbs",    val: GOALS.carbs,    unit: "g",    color: COLOR.carbs    },
            { label: "Fat",      val: GOALS.fat,      unit: "g",    color: COLOR.fat      },
          ].map(({ label, val, unit, color }) => (
            <div key={label} style={{
              display: "flex", justifyContent: "space-between", alignItems: "center",
              padding: "12px 14px", background: color.light, borderRadius: "var(--radius-md)", marginBottom: 10
            }}>
              <span style={{ fontWeight: 500, fontSize: 14, color: color.text }}>{label}</span>
              <span style={{ fontWeight: 500, fontSize: 16, color: color.text }}>{val} {unit}</span>
            </div>
          ))}
          <div style={{ fontSize: 12, color: "var(--color-text-tertiary)", marginTop: 8 }}>
            Based on Australian dietary guidelines for adolescents.
          </div>
        </div>
      )}
    </div>
  );
}

ReactDOM.createRoot(document.getElementById("root")).render(<App />);
